ROOT CAUSE: intake_node returned hardcoded 'infra' for all queries.

FIX (in app.py):
  # Before (broken):
  def intake_node(state):
      return {"intent": "infra"}  # BUG: hardcoded

  # After (correct):
  def intake_node(state):
      q = state["query"].lower()
      if any(w in q for w in ["train","gradient","nccl",...]):
          return {"intent": "training"}
      ...
